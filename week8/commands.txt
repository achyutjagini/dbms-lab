SELECT Fname,Minit,Lname
FROM EMPLOYEE
WHERE(SELECT COUNT(*) FROM DEPENDENT WHERE Essn=Ssn)>=2;


2.
SELECT Fname,Gender
FROM EMPLOYEE
WHERE Ssn=(
    SELECT Essn
    FROM DEPENDENT
    WHERE Dependent_name=Fname AND DEPENDENT.Gender=EMPLOYEE.Gender);


3.
SELECT Fname,Minit,Lname,Dno
FROM EMPLOYEE
WHERE Salary>ALL(SELECT Salary FROM EMPLOYEE WHERE dno=5);

4.
SELECT Fname,Minit,Lname,Ssn
FROM EMPLOYEE
WHERE NOT EXISTS(
   SELECT Essn FROM DEPENDENT
   WHERE DEPENDENT.Essn=EMPLOYEE.Ssn);


5.
SELECT Fname,Minit,Lname,Ssn FROM EMPLOYEE WHERE EXISTS(
SELECT Essn FROM DEPENDENT WHERE(
    DEPENDENT.Essn=EMPLOYEE.Ssn AND DEPENDENT.Essn IN
    (SELECT Mgr_ssn FROM DEPARTMENT)
)
);

6.
SELECT Fname,Minit,Lname,Dno,Address
FROM EMPLOYEE
JOIN DEPARTMENT ON DEPARTMENT.Dnumber=EMPLOYEE.Dno AND Dname='Research';
